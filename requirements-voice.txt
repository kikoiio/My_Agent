# requirements-voice.txt — 语音硬件依赖
# 安装: pip install -r requirements-voice.txt
#
# 在安装 requirements.txt 之后再安装此文件。
# 所有包均可在 Windows 11 / Ubuntu 22.04 + Python 3.11/3.12 上安装。
# Python 3.14 暂未全量支持（faster-whisper / resemblyzer 依赖 torch/ctranslate2 wheel）。

# ---- STT（语音转文字） ----
faster-whisper>=1.0       # CTranslate2 加速 Whisper，base 模型中文准确率高

# ---- 麦克风采集 ----
sounddevice>=0.4.6        # PortAudio 封装，Windows/Linux 均可用

# ---- TTS（文字转语音） ----
edge-tts>=6.1             # 微软 Edge TTS，免费，中文质量高，无需 API key
miniaudio>=1.2            # MP3/WAV 解码 → PCM，供 sounddevice 播放

# ---- 人脸识别 ----
insightface>=0.7          # buffalo_l 模型，首次运行自动下载 (~300 MB)
onnxruntime>=1.16         # insightface 推理后端（CPU 模式）
opencv-python-headless>=4.8  # 图像读取/解码（无 GUI 依赖）

# ---- 声纹验证 ----
resemblyzer>=0.1.1        # GE2E 预训练模型，17 MB，无需训练
